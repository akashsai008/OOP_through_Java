import java.util.*;
public class Student {
    int rollno;
           String name;
           double marks;
           Student(int rollno,String name,double marks){
               this.rollno=rollno;
               this.name=name;
               this.marks=marks;
              
           }
            Student (Student v){
                this.rollno=v.rollno;
                this.name=v.name;
                this.marks=v.marks;
            }
    public static void main(String[] args) {
          
       Student s1=new Student(1,"akash",69.69);
       Student s2=new Student(s1);
       s2.name="varma";
       
       System.out.println(s2.name+" "+s1.name);
    }
}
